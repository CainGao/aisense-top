单体架构优势：
✅ 开发简单：一个项目，统一管理
✅ 部署容易：一个二进制文件
✅ 调试方便：所有代码在一起
✅ 性能更好：无网络调用开销
✅ 成本更低：只需一台服务器
✅ 适合MVP：快速迭代验证

1.2 为什么选择单体架构

单体架构 vs 微服务架构

对比项 单体架构 微服务架构
开发复杂度 ✅ 简单（一个项目） ❌ 复杂（多个服务）
部署难度 ✅ 容易（一个二进制） ❌ 困难（多个容器）
调试效率 ✅ 高效（统一日志） ❌ 低效（分布式追踪）
运维成本 ✅ 低（一台服务器） ❌ 高（多台服务器）
性能 ✅ 好（无网络开销） ❌ 一般（网络调用）
适合阶段 ✅ MVP、初创期 ❌ 大规模、成熟期

结论：对于小说出海MVP项目，单体架构是最优选择！

3.2.1 main.go（HTTP服务入口）

package main

import (
    "context"
    "log"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    "time"
    
    "novel-platform/internal/config"
    "novel-platform/internal/handlers"
    "novel-platform/internal/middleware"
    "novel-platform/internal/repository"
    "novel-platform/internal/services"
    "novel-platform/pkg/cache"
    "novel-platform/pkg/logger"
    
    "github.com/gin-gonic/gin"
)

func main() {
    // 加载配置
    cfg, err := config.Load()
    if err != nil {
        log.Fatalf("Failed to load config: %v", err)
    }
    
    // 初始化日志
    logger.Init(cfg.Log)
    
    // 初始化Redis
    if err := cache.InitRedis(cfg.Redis); err != nil {
        log.Fatalf("Failed to init redis: %v", err)
    }
    
    // 初始化数据库
    db, err := config.InitDB(cfg.Database)
    if err != nil {
        log.Fatalf("Failed to init database: %v", err)
    }
    
    // 初始化Repository层
    userRepo := repository.NewUserRepository(db)
    novelRepo := repository.NewNovelRepository(db)
    chapterRepo := repository.NewChapterRepository(db)
    orderRepo := repository.NewOrderRepository(db)
    
    // 初始化Service层
    userService := services.NewUserService(userRepo)
    novelService := services.NewNovelService(novelRepo, chapterRepo)
    chapterService := services.NewChapterService(chapterRepo)
    orderService := services.NewOrderService(orderRepo)
    paymentService := services.NewPaymentService(cfg.Payment)
    aiService := services.NewAIService(cfg.AI)
    
    // 初始化Handler层
    userHandler := handlers.NewUserHandler(userService)
    novelHandler := handlers.NewNovelHandler(novelService)
    chapterHandler := handlers.NewChapterHandler(chapterService)
    orderHandler := handlers.NewOrderHandler(orderService, paymentService)
    
    // 创建路由
    router := gin.Default()
    
    // 配置全局中间件
    router.Use(middleware.CORS())
    router.Use(middleware.Logger())
    router.Use(middleware.RateLimit(cfg.RateLimit))
    router.Use(middleware.Recovery())
    
    // 注册路由
    api := router.Group("/api/v1")
    {
        // 用户相关
        users := api.Group("/users")
        {
            users.POST("/register", userHandler.Register)
            users.POST("/login", userHandler.Login)
            users.GET("/profile", middleware.Auth(), userHandler.Profile)
            users.PUT("/profile", middleware.Auth(), userHandler.UpdateProfile)
        }
        
        // 小说相关
        novels := api.Group("/novels")
        {
            novels.GET("", novelHandler.List)
            novels.GET("/:id", novelHandler.Detail)
            novels.GET("/:id/chapters", novelHandler.Chapters)
            novels.POST("", middleware.Auth(), middleware.Admin(), novelHandler.Create)
            novels.PUT("/:id", middleware.Auth(), middleware.Admin(), novelHandler.Update)
            novels.DELETE("/:id", middleware.Auth(), middleware.Admin(), novelHandler.Delete)
        }
        
        // 章节相关
        chapters := api.Group("/chapters")
        {
            chapters.GET("/:id", middleware.Auth(), chapterHandler.Detail)
            chapters.POST("/:id/read", middleware.Auth(), chapterHandler.MarkAsRead)
        }
        
        // 订单相关
        orders := api.Group("/orders")
        {
            orders.Use(middleware.Auth())
            orders.POST("", orderHandler.Create)
            orders.GET("", orderHandler.List)
            orders.GET("/:id", orderHandler.Detail)
        }
        
        // 支付相关
        payment := api.Group("/payment")
        {
            payment.Use(middleware.Auth())
            payment.POST("/create", paymentHandler.CreatePayment)
            payment.POST("/callback", paymentHandler.Callback)
        }
        
        // 用户书架
        bookshelf := api.Group("/bookshelf")
        {
            bookshelf.Use(middleware.Auth())
            bookshelf.GET("", novelHandler.Bookshelf)
            bookshelf.POST("/:id", novelHandler.AddToBookshelf)
            bookshelf.DELETE("/:id", novelHandler.RemoveFromBookshelf)
        }
        
        // 搜索
        search := api.Group("/search")
        {
            search.GET("", novelHandler.Search)
        }
    }
    
    // 健康检查
    router.GET("/health", func(c *gin.Context) {
        c.JSON(http.StatusOK, gin.H{
            "status": "ok",
            "timestamp": time.Now().Unix(),
        })
    })
    
    // 启动HTTP服务
    srv := &http.Server{
        Addr:    ":" + cfg.Server.Port,
        Handler: router,
    }
    
    // 优雅关闭
    go func() {
        if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
            log.Fatalf("Failed to start server: %v", err)
        }
    }()
    
    log.Printf("Server started on port %s", cfg.Server.Port)
    
    // 等待中断信号
    quit := make(chan os.Signal, 1)
    signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
    <-quit
    
    log.Println("Shutting down server...")
    
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()
    
    if err := srv.Shutdown(ctx); err != nil {
        log.Fatal("Server forced to shutdown:", err)
    }
    
    log.Println("Server exited")
}

3.2.2 novel_service.go（小说服务）

package services

import (
    "context"
    "errors"
    "novel-platform/internal/models"
    "novel-platform/internal/repository"
    
    "github.com/google/uuid"
)

type NovelService struct {
    novelRepo   *repository.NovelRepository
    chapterRepo *repository.ChapterRepository
    aiService   *AIService
}

func NewNovelService(
    novelRepo *repository.NovelRepository,
    chapterRepo *repository.ChapterRepository,
) *NovelService {
    return &NovelService{
        novelRepo:   novelRepo,
        chapterRepo: chapterRepo,
    }
}

// SetAIService 设置AI服务（依赖注入）
func (s *NovelService) SetAIService(aiService *AIService) {
    s.aiService = aiService
}

// CreateNovel 创建小说
func (s *NovelService) CreateNovel(ctx context.Context, req *CreateNovelRequest) (*models.Novel, error) {
    // 验证参数
    if err := req.Validate(); err != nil {
        return nil, err
    }
    
    // 创建小说
    novel := &models.Novel{
        ID:                uuid.New(),
        Title:             req.Title,
        Author:            req.Author,
        Category:          req.Category,
        Tags:              req.Tags,
        Description:       req.Description,
        Status:            "ongoing",
        FreeChapters:      req.FreeChapters,
        PricePerChapter:   req.PricePerChapter,
        SubscriptionPrice: req.SubscriptionPrice,
        UpdateFrequency:   req.UpdateFrequency,
        ChaptersPerDay:    req.ChaptersPerDay,
        TemplateID:        req.TemplateID,
        GenerationConfig:  req.GenerationConfig,
    }
    
    // 保存到数据库
    if err := s.novelRepo.Create(ctx, novel); err != nil {
        return nil, err
    }
    
    // 生成初始章节大纲
    if s.aiService != nil {
        if err := s.generateChapterOutlines(ctx, novel); err != nil {
            return nil, err
        }
    }
    
    return novel, nil
}

// UpdateNovel 更新小说

    }

        novel.Description = req.Description
    }
    if req.Status != "" {
        novel.Status = req.Status
    }
    
    return s.novelRepo.Update(ctx, novel)
}

// generateChapterOutlines 生成章节大纲
func (s *NovelService) generateChapterOutlines(ctx context.Context, novel *models.Novel) error {
    // 调用AI服务生成大纲
    outlines, err := s.aiService.GenerateChapterOutlines(ctx, novel)
    if err != nil {
        return err
    }
    
    // 保存章节大纲
    for _, outline := range outlines {
        chapter := &models.Chapter{
            NovelID:          novel.ID,
            ChapterNumber:    outline.ChapterNumber,
            Title:            outline.Title,
            Content:          "", // 内容稍后生成
            GenerationStatus: "pending",
        }
        
        if err := s.chapterRepo.Create(ctx, chapter); err != nil {
            return err
        }
    }
    
    return nil
}

// GenerateDailyChapters 每日生成章节（定时任务调用）
func (s *NovelService) GenerateDailyChapters(ctx context.Context) error {
    // 查询所有需要更新的小说
    novels, err := s.novelRepo.GetOngoing(ctx)
    if err != nil {
        return err
    }
    
    for _, novel := range novels {
        // 检查是否需要生成新章节
        if s.shouldGenerateChapter(novel) {
            // 生成章节
            if err := s.generateChapter(ctx, novel); err != nil {
                log.Printf("Failed to generate chapter for novel %s: %v", novel.ID, err)
                continue
            }
        }
    }
    
    return nil
}

// shouldGenerateChapter 判断是否需要生成章节
func (s *NovelService) shouldGenerateChapter(novel *models.Novel) bool {
    if novel.Status != "ongoing" {
        return false
    }
    
    // 检查更新频率
    switch novel.UpdateFrequency {
    case "daily":
        return true
    case "weekly":
        // 检查是否是每周的更新日
        return time.Now().Weekday() == time.Monday
    case "monthly":
        // 检查是否是每月的更新日
        return time.Now().Day() == 1
    default:
        return false
    }
}

// generateChapter 生成单章
func (s *NovelService) generateChapter(ctx context.Context, novel *models.Novel) error {
    // 获取下一个章节号
    nextChapterNum := novel.TotalChapters + 1
    
    // 获取前5章作为上下文
    previousChapters, err := s.chapterRepo.GetRecent(ctx, novel.ID, 5)
    if err != nil {
        return err
    }
    
    // 调用AI生成内容
    content, err := s.aiService.GenerateChapterContent(ctx, novel, nextChapterNum, previousChapters)
    if err != nil {
        return err
    }
    
    // 创建章节
    chapter := &models.Chapter{
        NovelID:          novel.ID,
        ChapterNumber:    nextChapterNum,
        Title:            fmt.Sprintf("Chapter %d", nextChapterNum),
        Content:          content,
        WordCount:        len(strings.Fields(content)),
        IsFree:           nextChapterNum <= novel.FreeChapters,
        GenerationStatus: "completed",
        PublishedAt:      timePtr(time.Now()),
    }
    
    // 保存章节
    if err := s.chapterRepo.Create(ctx, chapter); err != nil {
        return err
    }
    
    // 更新小说统计
    novel.TotalChapters++
    novel.TotalWords += chapter.WordCount
    novel.LastChapterAt = timePtr(time.Now())
    
    return s.novelRepo.Update(ctx, novel)
}

import (

type AIService struct {
    apiKey  string
    baseURL string
    client  *http.Client
}

func NewAIService(cfg *config.AIConfig) *AIService {
    return &AIService{
        apiKey:  cfg.APIKey,
        baseURL: cfg.BaseURL,
        client: &http.Client{
            Timeout: 60 * time.Second,
        },
    }
}

// GenerateChapterOutlines 生成章节大纲
func (s *AIService) GenerateChapterOutlines(ctx context.Context, novel *models.Novel) ([]*ChapterOutline, error) {
    prompt := fmt.Sprintf(`
Generate chapter outlines for an English novel with the following details:

Title: %s
Category: %s
Description: %s
Total Chapters: %d

Please generate chapter titles and brief descriptions for all chapters.
Format: JSON array with chapter_number and title fields.
`, novel.Title, novel.Category, novel.Description, novel.ChaptersPerDay*30)
    
    response, err := s.callGLM4(ctx, prompt)
    if err != nil {
        return nil, err
    }
    
    var outlines []*ChapterOutline
    if err := json.Unmarshal([]byte(response), &outlines); err != nil {
        return nil, err
    }
    
    return outlines, nil
}

// GenerateChapterContent 生成章节内容
func (s *AIService) GenerateChapterContent(ctx context.Context, novel *models.Novel, chapterNum int, previousChapters []*models.Chapter) (string, error) {
    // 构建上下文
    context := s.buildContext(novel, previousChapters)
    
    prompt := fmt.Sprintf(`
You are writing an English novel. Here's the context:

Novel Title: %s
Category: %s
Chapter %d

Previous Context:
%s

Write a compelling chapter of approximately 3000 words. Focus on:
1. Engaging narrative
2. Character development
3. Plot advancement
4. Vivid descriptions
5. Natural dialogue

Write the chapter content now:
`, novel.Title, novel.Category, chapterNum, context)
    
    content, err := s.callGLM4(ctx, prompt)
    if err != nil {
        return "", err
    }
    
    return content, nil
}

// callGLM4 调用GLM-4 API
func (s *AIService) callGLM4(ctx context.Context, prompt string) (string, error) {
    reqBody := map[string]interface{}{
        "model": "glm-4",
        "messages": []map[string]string{
            {
                "role":    "user",
                "content": prompt,
            },
        },
        "temperature": 0.8,
        "max_tokens":  4000,
    }
    
    body, err := json.Marshal(reqBody)
    if err != nil {
        return "", err
    }
    
    req, err := http.NewRequestWithContext(ctx, "POST", s.baseURL+"/chat/completions", bytes.NewReader(body))
    if err != nil {
        return "", err
    }
    
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("Authorization", "Bearer "+s.apiKey)
    
    resp, err := s.client.Do(req)
    if err != nil {
        return "", err
    }
    defer resp.Body.Close()
    
    if resp.StatusCode != http.StatusOK {
        bodyBytes, _ := io.ReadAll(resp.Body)
        return "", fmt.Errorf("API call failed: %s", string(bodyBytes))
    }
    
    var response struct {
        Choices []struct {
            Message struct {
                Content string `json:"content"`
            } `json:"message"`
        } `json:"choices"`
    }
    
    if err := json.NewDecoder(resp.Body).Decode(&response); err != nil {
        return "", err
    }
    
    if len(response.Choices) == 0 {
        return "", fmt.Errorf("no response from API")
    }
    
    return response.Choices[0].Message.Content, nil
}

// buildContext 构建上下文
func (s *AIService) buildContext(novel *models.Novel, previousChapters []*models.Chapter) string {
    context := fmt.Sprintf("Novel Overview: %s\n\n", novel.Description)
    
    // 添加最近5章的摘要
    start := len(previousChapters) - 5
    if start < 0 {
        start = 0
    }
    
    for i := start; i < len(previousChapters); i++ {
        chapter := previousChapters[i]
        summary := chapter.Content
        if len(summary) > 500 {
            summary = summary[:500] + "..."
        }
        context += fmt.Sprintf("Chapter %d: %s\n%s\n\n", 
            chapter.ChapterNumber, chapter.Title, summary)
    }
    
    return context
}

func main() {
    // 加载配置
    cfg, err := config.Load()
    if err != nil {
        log.Fatalf("Failed to load config: %v", err)
    }
    
    // 初始化
    logger.Init(cfg.Log)
    db, err := config.InitDB(cfg.Database)
    if err != nil {
        log.Fatalf("Failed to init database: %v", err)
    }
    
    // 初始化Repository和Service
    novelRepo := repository.NewNovelRepository(db)
    chapterRepo := repository.NewChapterRepository(db)
    aiService := services.NewAIService(cfg.AI)
    
    novelService := services.NewNovelService(novelRepo, chapterRepo)
    novelService.SetAIService(aiService)
    
    // 创建定时任务
    c := cron.New(cron.WithSeconds())
    
    // 任务1：每日生成小说章节（02:00）
    c.AddFunc("0 0 2 * * *", func() {
        log.Println("Starting daily novel generation task...")
        ctx, cancel := context.WithTimeout(context.Background(), 4*time.Hour)
        defer cancel()
        
        if err := novelService.GenerateDailyChapters(ctx); err != nil {
            log.Printf("Daily generation task failed: %v", err)
        }
    })
    
    // 任务2：数据统计（08:00）
    c.AddFunc("0 0 8 * * *", func() {
        log.Println("Starting daily statistics task...")
        ctx, cancel := context.WithTimeout(context.Background(), 30*time.Minute)
        defer cancel()
        
        if err := novelService.GenerateDailyReport(ctx); err != nil {
            log.Printf("Daily statistics task failed: %v", err)
        }
    })
    
    // 任务3：质量监控（每小时）
    c.AddFunc("0 0 * * * *", func() {
        log.Println("Starting hourly quality check...")
        ctx, cancel := context.WithTimeout(context.Background(), 10*time.Minute)
        defer cancel()
        
        if err := novelService.QualityCheck(ctx); err != nil {
            log.Printf("Quality check failed: %v", err)
        }
    })
    
    // 启动定时任务
    c.Start()
    log.Println("Cron jobs started...")
    
    // 阻塞主线程
    select {}
}
