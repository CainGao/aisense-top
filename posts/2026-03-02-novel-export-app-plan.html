📊 数据库设计（优化后）

核心表结构

1. novels（小说表）

CREATE TABLE novels (
    id UUID PRIMARY KEY,
    title VARCHAR(255) NOT NULL,  -- 英文标题
    author VARCHAR(100) DEFAULT 'AI',
    category VARCHAR(50),  -- Fantasy, Romance, etc.
    tags TEXT[],  -- 英文标签
    description TEXT,  -- 英文简介
    cover_url TEXT,
    status VARCHAR(20) DEFAULT 'ongoing',  -- ongoing, completed
    total_chapters INTEGER DEFAULT 0,
    total_words INTEGER DEFAULT 0,
    free_chapters INTEGER DEFAULT 20,
    price_per_chapter DECIMAL(10,2) DEFAULT 0.05,
    subscription_price DECIMAL(10,2) DEFAULT 9.99,
    view_count INTEGER DEFAULT 0,
    read_count INTEGER DEFAULT 0,
    favorite_count INTEGER DEFAULT 0,
    rating DECIMAL(3,2) DEFAULT 0,
    rating_count INTEGER DEFAULT 0,
    update_frequency VARCHAR(20) DEFAULT 'daily',
    chapters_per_day INTEGER DEFAULT 3,
    start_date TIMESTAMP,
    end_date TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

2. chapters（章节表）

CREATE TABLE chapters (
    id UUID PRIMARY KEY,
    novel_id UUID REFERENCES novels(id),
    chapter_number INTEGER NOT NULL,
    title VARCHAR(255),  -- 英文章节标题
    content TEXT NOT NULL,  -- 英文内容
    word_count INTEGER,
    is_free BOOLEAN DEFAULT false,
    price DECIMAL(10,2) DEFAULT 0.05,
    view_count INTEGER DEFAULT 0,
    generation_status VARCHAR(20) DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT NOW()
);

3. generation_tasks（生成任务表）

CREATE TABLE generation_tasks (
    id UUID PRIMARY KEY,
    novel_id UUID REFERENCES novels(id),
    task_type VARCHAR(50),
    status VARCHAR(20) DEFAULT 'pending',
    config JSONB,
    result JSONB,
    error_message TEXT,
    retry_count INTEGER DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP
);

4. cron_logs（定时任务日志表）

CREATE TABLE cron_logs (
    id UUID PRIMARY KEY,
    task_name VARCHAR(100),
    execution_time TIMESTAMP,
    status VARCHAR(20),
    novels_processed INTEGER,
    chapters_generated INTEGER,
    errors TEXT[],
    report TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);
